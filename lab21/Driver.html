import java.util.Date;
import java.util.Scanner;

public class Driver {

    public static Dog[] create3Dogs()
    {
        Scanner scan = new Scanner(System.in);
        Dog[] ar = new Dog[3];
        for (int i=0; i <3; ++i)
        {
            System.out.println("enter name, species, year, month, day for dog: ");
            ar[i] = new Dog (scan.next(), scan.next(), new Date(scan.nextInt(), scan.nextInt(), scan.nextInt()));
        }
        return ar;
    }
    private static void printDogs(Dog[] ar) {
        for (Dog d: ar)
        {
            System.out.println(d);
        }
        
    }
    private static void sortDogsByName(Dog[] ar) 
    {
        for (int i=0; i < ar.length-1; ++i)
        {
            for (int j = i+1; j < ar.length; ++j)
            {
                if (ar[i].getName().compareToIgnoreCase(ar[j].getName()) > 0)
                {
                    Dog temp = ar[i];
                    ar[i] = ar[j];
                    ar[j] = temp;
                }
            }
        }
    }
    public static void main(String[] args) {

        Dog[] ar = Driver.create3Dogs();
        Driver.printDogs(ar);
    }
}
